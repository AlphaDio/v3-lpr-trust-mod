# Bond Market Trust Tier Modifiers
# These modifiers affect loan interest rates based on country bond trust
# Lower trust = higher interest rates (penalty for risky borrowing)
# Higher trust = lower interest rates (reward for good credit)

# Very Low Trust (0-19): Severe interest penalty and greatly reduced credit capacity
bond_trust_tier_very_low = {
	country_loan_interest_rate_add = 0.05  # +5% interest rate
	building_cash_reserves_mult = -0.3  # -30% cash reserves for all buildings
}

# Low Trust (20-39): High interest penalty and reduced credit capacity
bond_trust_tier_low = {
	country_loan_interest_rate_add = 0.03  # +3% interest rate
	building_cash_reserves_mult = -0.15  # -15% cash reserves for all buildings
}

# Medium Trust (40-59): Moderate interest penalty and neutral credit capacity
bond_trust_tier_medium = {
	country_loan_interest_rate_add = 0.01  # +1% interest rate
	# No cash reserves modifier for medium trust - neutral effect
}

# High Trust (60-79): Interest bonus and increased credit capacity
bond_trust_tier_high = {
	country_loan_interest_rate_add = -0.01  # -1% interest rate
	building_cash_reserves_mult = 0.1  # +10% cash reserves for all buildings
}

# Very High Trust (80-100): Significant interest bonus and greatly increased credit capacity
bond_trust_tier_very_high = {
	country_loan_interest_rate_add = -0.03  # -3% interest rate
	building_cash_reserves_mult = 0.2  # +20% cash reserves for all buildings
}
