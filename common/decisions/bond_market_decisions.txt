# Bond Market Debug Decisions
# Debug decisions for testing the bond trust system

bond_market_show_trust = {
		icon = "gfx/interface/icons/decision_icons/generic_decision.dds"
		available = {
			always = yes
		}
		visible = {
			always = yes
		}
		effect = {
			if = {
				limit = {
					NOT = { has_variable = bond_trust }
				}
				set_variable = { name = bond_trust value = 60 }
			}
			country_event = bond_market_events.2
		}
}
