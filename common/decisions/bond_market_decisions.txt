# Bond Market Debug Decisions
# Debug decisions for testing the bond trust system

# Category for bond market debug decisions
bond_market_debug_category = {
		icon = debug_mode

		# Decision to show current bond trust
		bond_market_show_trust = {
			icon = debug_mode
			available = {
				is_debug_mode = yes
			}
			visible = {
				is_debug_mode = yes
			}

			# Initialize trust if missing
			effect = {
				if = {
					limit = {
						NOT = { has_country_variable = bond_trust }
					}
					set_country_variable = { name = bond_trust value = 60 }
				}
				country_event = bond_market_debug_show.1
			}
		}

		# Decision to add the bond market journal entry
		bond_market_add_journal_entry = {
			icon = debug_mode
			available = {
				is_debug_mode = yes
			}
			visible = {
				is_debug_mode = yes
			}

			effect = {
				add_journal_entry = {
					type = je_bond_market_debug
				}
			}
		}

		# Decision to set trust to 10 (very low)
		bond_market_set_trust_10 = {
			icon = debug_mode
			available = {
				is_debug_mode = yes
			}
			visible = {
				is_debug_mode = yes
			}

			effect = {
				set_country_variable = { name = bond_trust value = 10 }
				country_event = bond_market_debug_set.1
			}
		}

		# Decision to set trust to 60 (medium)
		bond_market_set_trust_60 = {
			icon = debug_mode
			available = {
				is_debug_mode = yes
			}
			visible = {
				is_debug_mode = yes
			}

			effect = {
				set_country_variable = { name = bond_trust value = 60 }
				country_event = bond_market_debug_set.1
			}
		}

		# Decision to set trust to 95 (very high)
		bond_market_set_trust_95 = {
			icon = debug_mode
			available = {
				is_debug_mode = yes
			}
			visible = {
				is_debug_mode = yes
			}

			effect = {
				set_country_variable = { name = bond_trust value = 95 }
				country_event = bond_market_debug_set.1
			}
		}
	}
}
