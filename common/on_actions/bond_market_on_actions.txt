# Bond Market On Actions
# This file contains hooks that trigger the monthly bond trust updates

# Monthly pulse for bond trust system
on_monthly_pulse_country = {
	delay = { days = { 1 5 } }

	events = {
		bond_market_events.1  # Trigger trust update event for every country
	}

	# Combined bond market random events
	# Total: 60% no event, 41% chance of various events
	random_events = {
		60 = 0  # 59% chance of no event
		25 = bond_market_events.2  # 25% chance of income bonus (medium+ trust)
		8 = bond_market_events.4   # 8% chance of union unrest (trust > 70)
		5 = bond_market_events.3   # 5% chance of capitalist unrest (trust < 50)
		3 = bond_market_events.5   # 3% chance of industrialist loyalty (trust >= 30)
	}
}

# Half-yearly pulse for bond rating grade changes
on_half_yearly_pulse_country = {
	events = {
		bond_market_events.6  # AAA rating (95-100)
		bond_market_events.7  # AA rating (85-94)
		bond_market_events.8  # A rating (75-84)
		bond_market_events.9  # B rating (65-74)
		bond_market_events.10 # C rating (50-64)
		bond_market_events.11 # D rating (25-49)
		bond_market_events.12 # F rating (0-24)
	}
}
