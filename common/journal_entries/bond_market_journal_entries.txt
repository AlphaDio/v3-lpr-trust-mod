# ============================================================================
# THE BOND MARKET JOURNAL ENTRY
# ============================================================================
# This journal entry provides debugging information for the bond market system.
# It displays current bond trust value, tier, and economic indicators to help
# players and developers understand how the bond market mechanics are working.
# ============================================================================

# Journal entry that displays bond market debug information
je_bond_market_debug = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	# Groups this entry under Internal Affairs in the journal UI
	group = je_group_internal_affairs

	# Always show this entry for player countries
	is_shown_when_inactive = {
		is_player = yes
	}

	# Always available to player countries
	possible = {
		is_player = yes
	}

	# Never completes - this is a permanent debugging entry
	complete = {
		always = no
	}

	# Display the current bond market debug information
	status_desc = {
		desc = je_bond_market_debug_status
	}
}

# Journal entry for when country is in default
je_bond_market_default = {
	icon = "gfx/interface/icons/event_icons/event_bankruptcy.dds"

	# Groups this entry under Internal Affairs in the journal UI
	group = je_group_internal_affairs

	# Show this entry when inactive
	is_shown_when_inactive = {
		is_player = yes
	}

	# Available when player is in default
	possible = {
		is_player = yes
		in_default = yes
	}

	# Complete when no longer in default
	complete = {
		NOT = { in_default = yes }
	}

	# Display default status information
	status_desc = {
		desc = je_bond_market_default_status
	}

	# Weekly pulse to trigger recovery choices
	on_weekly_pulse = {
		random_events = {
			90 = 0  # 90% chance of no event
			10 = bond_market_events.6  # 10% chance of default recovery choice
		}
	}
}
